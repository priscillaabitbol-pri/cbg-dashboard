🔗 Conectar Planilha Google Sheets
Cole um único link e o dashboard lê todas as abas automaticamente — incluindo as novas que você criar.
1
Abra o Google Apps Script da sua planilha
Na sua planilha: clique em Extensões → Apps Script. Uma nova aba vai abrir com o editor de código.
2
Cole este código (apague tudo que tiver lá e cole isso):
function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheets = ss.getSheets();
  var data = {};
  // Abas a ignorar (controle mensal, modelos etc)
  var ignorar = ['controle mensal','modelo','config','base','totais','resumo'];
  sheets.forEach(function(sheet) {
    var name = sheet.getName();
    var nameLower = name.toLowerCase().normalize('NFD').replace(/[\u0300-\u036f]/g,'');
    var skip = ignorar.some(function(i){ return nameLower.indexOf(i) >= 0; });
    if (skip) return;
    var rows = sheet.getDataRange().getValues();
    if (rows.length > 1) {
      data[name] = rows.slice(1).filter(function(r) {
        return r[4] && String(r[4]).trim(); // filtra por razão social (col E)
      });
    }
  });
  var cb = e && e.parameter && e.parameter.callback;
  var json = JSON.stringify(data);
  if (cb) {
    return ContentService.createTextOutput(cb + '(' + json + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return ContentService.createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}
3
Implante o script como Web App
Clique em Implantar → Nova implantação
Tipo: App da Web
Executar como: Eu (seu e-mail)
Quem pode acessar: Qualquer pessoa
Clique em Implantar → autorize → copie o link gerado
